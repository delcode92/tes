import numpy as np
import cv2 as cv
import time

# cap = cv.VideoCapture('rtsp://192.168.100.69:4747')
# cap = cv.VideoCapture('http://192.168.100.69:4747/video?640x480')
# cap = cv.VideoCapture('http://192.168.168.69:4747/video?640x480')
cap = cv.VideoCapture(0)
if not cap.isOpened():
    print("Cannot open camera")
    exit()
while True:
    # Capture frame-by-frame
    ret, frame = cap.read()
    # if frame is read correctly ret is True
    if not ret:
        print("Can't receive frame (stream end?). Exiting ...")
        break
    # Our operations on the frame come here
    color = cv.cvtColor(frame, cv.COLOR_BGR2BGRA)
    # Display the resulting frame
    cv.imshow('frame', color)
    # time.sleep(0.04)

    if cv.waitKey(1) == ord('q'):
        cv.imwrite('c1.png',color)
        break
    
    cv.waitKey(30)

# When everything done, release the capture
cap.release()
cv.destroyAllWindows()
